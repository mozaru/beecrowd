``` 
[principal:]
	lerInt(n)
	para it,1,n,1
		lerLinha(linha,10000)
		qtd <- comprimento(linha)		
		enquanto ascii(obterLetra(linha, qtd-1)) <= 13
		  qtd <- qtd - 1
		fimenquanto
		i <- truncar(qtd / 2)-1
		enquanto i>=0
			l <- obterLetra(linha, i)
			escreverNaLinha("{l}")
			i <- (i - 1)
		fimenquanto
		j <- truncar(qtd/2)
		i <- (qtd - 1)
		enquanto i>=j
			l <- obterLetra(linha, i)
			escreverNaLinha("{l}")
			i <- (i - 1)
		fimenquanto
		escrever("")
	fimpara
```
